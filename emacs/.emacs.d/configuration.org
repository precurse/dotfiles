* Use evil
    #+BEGIN_SRC emacs-lisp
    (use-package evil
    :config

    (use-package evil-surround
    :config
    (global-evil-surround-mode 1))

    (use-package evil-org
    :after org
    :config
    (add-hook 'org-mode-hook 'evil-org-mode)

    )
